function checkViral(){ let topic=document.getElementById("topic").value.toLowerCase() let score=30 if(topic.includes("ai")) score+=40 if(topic.includes("youtube")) score+=20 if(topic.includes("money")) score+=20 if(topic.includes("automation")) score+=20 if(topic.includes("chatgpt")) score+=30 if(score>100) score=100 let competition="Medium" if(score>70) competition="Low" if(score<40) competition="High" let platform="YouTube" if(score>75) platform="YouTube Shorts" if(score>85) platform="TikTok" document.getElementById("result").innerHTML= `

${topic}

Viral Score: ${score}/100

Competition: ${competition}

Best Platform: ${platform}

` }
